114  void realsense_grabcut () {
115  if (!l_rs_cam) {
116  // we need a valid link to realsense cam sofa component
117  std::cerr <<
118  "(RealSenseGrabCut) link to realsense cam component is broken" <<
119  std::endl ;
120  return ;
121  }
122 
123  // get depth frame
124  rs2::colorizer rizer ;
125  rizer.set_option(RS2_OPTION_COLOR_SCHEME, 2);
126  rs2::depth_frame depth = *l_rs_cam->depth ;
127  rs2::frame bw_depth = depth.apply_filter(rizer) ;
128 
129  // get near image mask
130  cv::Mat near = frame_to_mat(bw_depth) ;
131  cv::cvtColor(near, near, cv::COLOR_BGR2GRAY);
132  create_mask_from_depth(near, d_near_thr.getValue(), cv::THRESH_BINARY);
133 
134  // get far image mask
135  cv::Mat far = frame_to_mat(bw_depth) ;
136  cv::cvtColor(far, far, cv::COLOR_BGR2GRAY) ;
137  far.setTo(255, far==0) ;
138  create_mask_from_depth(far, d_far_thr.getValue(), cv::THRESH_BINARY_INV);
139 
140  // create mask
141  cv::Mat mask;
142  mask.create(near.size(), CV_8UC1);
143  mask.setTo(cv::Scalar::all(cv::GC_BGD)) ; // Set "background" as default guess
144  mask.setTo(cv::GC_PR_BGD, far==0) ; // Relax this to "probably background" for pixels outside "far" region
145  mask.setTo(cv::GC_PR_FGD, near==255) ; // Set pixels within the "near" region to "foreground"
146 
147  // Run Grab-Cut algorithm:
148  cv::Mat bgModel, fgModel,
149  color_mat = d_image_in.getValue(),
150  depth_mat = d_depth_in.getValue() ;
151 
152  // get ROI
153 // defaulttype::Vec4i & roi = *d_rectangle.beginEdit() ;
154 // cv::Rect2d rect ;
155 // if (roi[0] == 0 && roi[1] == 0 && roi[2] == 0 && roi[3] == 0) {
156 // // rect = cv::selectROI(color_mat);
157 // cv::namedWindow("rect") ;
158 // rect = opencvplugin::utils::mouseevents::rectangleDrawer(
159 // "rect", d_image_in.getValue()) ;
160 // roi[0] = rect.x ; roi[1] = rect.y ;
161 // roi[2] = rect.width ; roi[3] = rect.height ;
162 // std::cout << roi << std::endl ;
163 // } else {
164 // rect = cv::Rect2d(roi[0],roi[1],roi[2],roi[3]) ;
165 // }
166 // d_rectangle.endEdit();
167 
168  cv::grabCut(
169  color_mat,
170  mask,
171  rect,
172  bgModel,
173  fgModel,
174  1,
175  cv::GC_INIT_WITH_MASK
176  );
177 
178  // Extract foreground pixels based on refined mask from the algorithm
179  cv::Mat maskimg , & imageDest = *d_image_out.beginEdit(), imgtmp, imgtmp2, & depthDest = *d_depth_out.beginEdit() ;
180  cv::compare(mask,cv::GC_PR_FGD,maskimg,cv::CMP_EQ);
181 
182  if (depth_mat.size() == maskimg.size()) {
183  depth_mat.copyTo(imgtmp2, maskimg);
184  depthDest = maskimg.clone() ;
185  }
186 
187  color_mat.copyTo(imgtmp, maskimg); // imageDest is the output
188  imageDest = imgtmp.clone() ;
189 
190  d_image_out.endEdit();
191  d_depth_out.endEdit();
192  }

196  void create_mask_from_depth (cv::Mat& depth, int thresh, cv::ThresholdTypes type) {
197  #define EROSION_KERNEL_SIZE cv::Size(3, 3)
198  cv::Mat
199  erode_less = cv::getStructuringElement(cv::MORPH_RECT, EROSION_KERNEL_SIZE),
200  erode_more = cv::getStructuringElement(cv::MORPH_RECT, EROSION_KERNEL_SIZE*2) ;
201  cv::threshold(depth, depth, thresh, 255, type);
202  cv::dilate(depth, depth, erode_less);
203  cv::erode(depth, depth, erode_more);
204  }
