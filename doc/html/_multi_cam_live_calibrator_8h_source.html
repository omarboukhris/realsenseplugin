110  void checkForCorners() {
111  if (calibimage1.size() == 30 || !activated) {
112  activated = false ;
113  return ; // 30 should be enough, can be changed
114  }
115  defaulttype::Vector2 bsize = d_chessboardsize.getValue() ;
116  cv::Size boardsize = cv::Size(bsize[0],bsize[1]) ;
117 
118  std::vector<cv::Point2f> corners1, corners2 ;
119  cv::Mat img1 = d_img1.getValue().getImage(),
120  img2 = d_img2.getValue().getImage(),
121  grey1, grey2 ;
122  if (img1.rows*img1.cols == 0 || img2.rows*img2.cols == 0 ) {
123  return ;
124  }
125  cv::cvtColor(img1, grey1, cv::COLOR_BGR2GRAY) ;
126  cv::cvtColor(img2, grey2, cv::COLOR_BGR2GRAY) ;
127  bool gotCorners = getCorners(boardsize, grey1, corners1, grey2, corners2);
128  if (gotCorners) {
129  calibimage1.push_back(grey1);
130  calibimage2.push_back(grey2);
131  cv::drawChessboardCorners(grey1,boardsize,corners1,true) ;
132  cv::drawChessboardCorners(grey2,boardsize,corners2,true) ;
133  d_img1out.setValue(grey1);
134  d_img2out.setValue(grey2);
135  }
136  }
