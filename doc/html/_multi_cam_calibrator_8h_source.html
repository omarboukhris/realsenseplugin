69 static void calcChessboardCorners(cv::Size boardSize, float squareSize, std::vector<cv::Point3f>& corners)
70 {
71  corners.resize(0);
72 
73  for( int i = 0; i < boardSize.height; i++ )
74  for( int j = 0; j < boardSize.width; j++ )
75  corners.push_back(cv::Point3f(float(j*squareSize),
76  float(i*squareSize), 0));

144  int loadImages (const std::string & path, std::vector<cv::Mat> & output) {
145  output.clear() ;
146  if (helper::system::FileSystem::isDirectory(path)) {
147  std::vector<std::string> png_images ;
148  helper::system::FileSystem::listDirectory(path, png_images, std::string(".png")) ;
149  std::sort(png_images.begin(), png_images.end()) ;
150  for (auto p : png_images) {
151  cv::Mat image = cv::imread(path + "/" + p), grey ;
152  cv::cvtColor(image, grey, cv::COLOR_BGR2GRAY) ;
153  output.push_back(grey);
154  }
155  return output.size() ;
156  }
157  return 0 ;
158  }

169  bool getCorners(
170  const cv::Size boardsize,
171  cv::Mat img1, std::vector<cv::Point2f> & corners1,
172  cv::Mat img2, std::vector<cv::Point2f> & corners2)
173  {
174  bool found1 = cv::findChessboardCorners(img1, boardsize, corners1, cv::CALIB_CB_ADAPTIVE_THRESH | cv::CALIB_CB_FILTER_QUADS) ;
175  bool found2 = cv::findChessboardCorners(img2, boardsize, corners2, cv::CALIB_CB_ADAPTIVE_THRESH | cv::CALIB_CB_FILTER_QUADS) ;
176  if (found1 && found2) {
177  cv::cornerSubPix(
178  img1, corners1,
179  cv::Size(5,5), cv::Size(-1,-1),
180  cv::TermCriteria(CV_TERMCRIT_EPS|CV_TERMCRIT_ITER, 42, 1e-2)) ;
181  cv::cornerSubPix(
182  img2, corners2,
183  cv::Size(5,5), cv::Size(-1,-1),
184  cv::TermCriteria(CV_TERMCRIT_EPS|CV_TERMCRIT_ITER, 42, 1e-2)) ;
185  img_points1.push_back(corners1);
186  img_points2.push_back(corners2);
187  add_obj_points();
188  return true ;
189  }
190  return false ;
191  }

280  void load_all()
281  {
282  img_points1.clear() ; img_points2.clear();
283  int i1 = loadImages(d_calibcam1.getValue(),calibimage1) ;
284  int i2 = loadImages(d_calibcam2.getValue(),calibimage2) ;
285  if (i1 != i2 || i1 * i2 == 0) {
286  std::cerr << "(MultiCamCalibrator) numbers of images is different, check input folders" << std::endl ;
287  return ;
288  }
289  std::cout << "loading done : got " << calibimage1.size() << " images" << std::endl ;
290 
291  defaulttype::Vector2 bsize = d_chessboardsize.getValue() ;
292  cv::Size boardsize = cv::Size(bsize[0],bsize[1]) ;
293  get_image_points(boardsize, calibimage1, calibimage2);
294  std::cout << "got img points : " << img_points1.size() << std::endl ;
295  }

301  void process () {
302 // load images
303  if (checkImgPointsValid()) {
304  std::cerr << "(MultiCamCalibrator) Error computing image points" << std::endl ;
305  return ;
306  }
307  std::cout << "(MultiCamCalibrator) Valid image points" << std::endl ;
308 
309 // call camera calibration then
310  std::cout << "(MultiCamCalibrator) Intrinsic calibration" << std::endl ;
311  cv::Mat cm1, d1, cm2, d2 ;
312  cv::Size imgsize = calibimage1[0].size() ;
313  calibratemono(cm1, d1, img_points1, imgsize);
314  calibratemono(cm2, d2, img_points2, imgsize);
315  std::cout << "(MultiCamCalibrator) Intrinsic calibration done" << std::endl ;
316 
317 // call cv::stereoCalibrate()
318 // output is extrinsic R|T matrix
319  std::cout << "(MultiCamCalibrator) Extrinsic calibration" << std::endl ;
320  cv::Mat R, T, E, F;
321  cv::stereoCalibrate(
322  obj_points,
323  img_points1, img_points2,
324  cm1, d1,
325  cm2, d2,
326  imgsize,
327  R, T, E, F,
328  cv::CALIB_SAME_FOCAL_LENGTH | cv::CALIB_ZERO_TANGENT_DIST,
329  cv::TermCriteria(CV_TERMCRIT_ITER|CV_TERMCRIT_EPS, 100, 1e-5));
330  std::cout << "(MultiCamCalibrator) Done calibrating : " << R << " " << T << std::endl ;
331 
332  write2output(T, R);
333  }
