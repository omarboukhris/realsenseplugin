143  void saveFrameToStream() {
144  RealSenseDistFrame distframe = d_distframe.getValue() ;
145  RealSenseDistFrame::RealSenseDistStruct diststruct = distframe.getFrame();
146 
147  // write width and height

156  filestream

161  void saveFrame () {
162  updateFileStream();
163  if (filestream == nullptr) {
164  std::cerr << "stream is unopened. check rights on file" << std::endl ;
165  return ;
166  }
167 
168  saveFrameToStream();
169 
170  if (++frame_count >= d_fpf.getValue()) {
171  std::fclose(filestream) ;
172  filestream = nullptr ;
173  }
174  }

176  std::string processFileName () {
177  // reads file name from data and processes it
178  // (add digit before extension)
179  std::string
180  extension = d_filename.getValue() , // starts as whole filename but ends up as extension
181  delimiter = "." ;
182  std::string filename = extension.substr(0, extension.find_last_of(delimiter)) ;
183  extension.erase(0, extension.find_last_of(delimiter)) ;
184  return filename + std::to_string(++file_id) + extension ;
185  }
