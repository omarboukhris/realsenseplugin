9 // Convert rs2::frame to cv::Mat
10 inline cv::Mat frame_to_mat(const rs2::frame& f)
11 {
12  using namespace cv;
13  using namespace rs2;
14 
15  auto vf = f.as<video_frame>();
16  const int w = vf.get_width();
17  const int h = vf.get_height();
18 
19  if (f.get_profile().format() == RS2_FORMAT_BGR8)
20  {
21  return Mat(Size(w, h), CV_8UC3, (void*)f.get_data(), Mat::AUTO_STEP);
22  }
23  else if (f.get_profile().format() == RS2_FORMAT_RGB8)
24  {
25  auto r = Mat(Size(w, h), CV_8UC3, (void*)f.get_data(), Mat::AUTO_STEP);
26  cvtColor(r, r, COLOR_RGB2BGR);
27  return r;
28  }
29  else if (f.get_profile().format() == RS2_FORMAT_Z16)
30  {
31  return Mat(Size(w, h), CV_16UC1, (void*)f.get_data(), Mat::AUTO_STEP);
32  }
33  else if (f.get_profile().format() == RS2_FORMAT_Y8)
34  {
35  return Mat(Size(w, h), CV_8UC1, (void*)f.get_data(), Mat::AUTO_STEP);
36  }
37  else if (f.get_profile().format() == RS2_FORMAT_DISPARITY32)
38  {
39  return Mat(Size(w, h), CV_32FC1, (void*)f.get_data(), Mat::AUTO_STEP);
40  }
41 
42  throw std::runtime_error("Frame format is not supported yet!");
43 }

45 // Converts depth frame to a matrix of doubles with distances in meters
46 inline cv::Mat depth_frame_to_meters(const rs2::pipeline& pipe, const rs2::depth_frame& f)
47 {
48  using namespace cv;
49  using namespace rs2;
50 
51  Mat dm = frame_to_mat(f);
52  dm.convertTo(dm, CV_64F);
53  auto depth_scale = pipe.get_active_profile()
54  .get_device()
55  .first<depth_sensor>()
56  .get_depth_scale();
57  dm = dm * depth_scale;
58  return dm;
59 }
