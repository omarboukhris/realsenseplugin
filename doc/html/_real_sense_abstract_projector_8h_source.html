232  inline void deproject_image_offline () {
233  helper::AdvancedTimer::stepBegin("RS Deprojection offline") ;
234  cv::Mat depth_im = d_depth.getValue().getImage() ;
235  RealSenseDistFrame distframe = d_distframe.getValue() ;
236  RealSenseDistFrame::RealSenseDistStruct diststruct = distframe.getFrame() ;
237  int downSample = d_downsampler.getValue() ;
238 
239  if (diststruct._width != (size_t)(depth_im.cols/downSample) ||
240  diststruct._height != (size_t)(depth_im.rows/downSample)) {
241  //std::cerr << "(deprojector) check sizes" << std::endl ;
242  return ;
243  }
244  if (depth_im.cols * depth_im.rows == 0) {
245  return ;
246  }
247  m_pointcloud->clear();
248  writeOfflineToOutput(diststruct, depth_im, downSample);
249  helper::AdvancedTimer::stepBegin("RS Deprojection offline") ;
250  }

256  inline void deproject_image_online () {
257  helper::AdvancedTimer::stepBegin("RS Deprojection online") ;
258  // get intrinsics from link to rs-cam component
259  rs2::depth_frame depth = *l_rs_cam->depth ;
260  cam_intrinsics = depth.get_profile().as<rs2::video_stream_profile>().get_intrinsics() ;
261 
262  // get depth image, and downsampling value
263  int downSample = d_downsampler.getValue() ;
264  cv::Mat depth_im = d_depth.getValue().getImage() ;
265  RealSenseDistFrame::RealSenseDistStruct & diststruct = *d_distframe.beginEdit();
266  diststruct._width = depth_im.cols/downSample ;
267  diststruct._height = depth_im.rows/downSample ;
268  diststruct.frame = new float[
269  depth_im.cols/downSample *
270  depth_im.rows/downSample
271  ] ;
272  m_pointcloud->clear();
273  writeOnlineToOutput(depth, diststruct, depth_im, downSample);
274  d_distframe.endEdit() ;
275  helper::AdvancedTimer::stepEnd("RS Deprojection online") ;
276  }

297  inline void push_to_pointcloud(helper::vector<defaulttype::Vector3> & outpoints, size_t i, size_t j, int index, RealSenseDistFrame::RealSenseDistStruct& diststruct, float dist)
298  {
299  float
300  point3d[3] = {0.f, 0.f, 0.f},
301  point2d[2] = {static_cast<float>(i), static_cast<float>(j)};
302  rs2_deproject_pixel_to_point(
303  point3d,
304  &cam_intrinsics,
305  point2d,
306  dist
307  );
308  // set dist frame for exportation if needed
309  diststruct.frame[index] = dist ;
310 
311  // check for outliers
312  if (std::abs(point3d[0]) < 1e-4 ||
313  std::abs(point3d[1]) < 1e-4 ||
314  std::abs(point3d[2]) < 1e-4 ||
315  std::abs(point3d[0]) > 5 ||
316  std::abs(point3d[1]) > 5 ||
317  std::abs(point3d[2]) > 0.45) {
318  //invalid point
319  return ;
320  }
321 
322  pcl::PointXYZ pt = scalePoint(point3d) ;
323  m_pointcloud->push_back(pt);
324 
325  defaulttype::Vector3 point = defaulttype::Vector3(pt.x, pt.y, pt.z) ;
326  if (d_flip.getValue()) {
327  point = defaulttype::Vector3(pt.y, pt.x, - pt.z) ;
328  }
329  applyRotationTranslation(point);
330  outpoints.push_back(point) ;
331  }

333  void makeSyntheticVolume () {
334  int dense = d_densify.getValue() ;
335  if (dense <= 0) {
336  // do not densify
337  return ;
338  }
339  helper::vector<defaulttype::Vector3> & synth = *d_synthvolume.beginEdit();
340  synth.clear() ;
341  for (const auto & ptmp : *m_pointcloud) {
342  for (int i = 1 ; i<dense ; i++) {
343  pcl::PointXYZ pt = ptmp ;
344  defaulttype::Vector3 point = defaulttype::Vector3(pt.x, pt.y, pt.z - i*1e-2) ;
345  if (d_flip.getValue()) {
346  point = defaulttype::Vector3(pt.y, pt.x, - pt.z + i*1e-2) ;
347  }
348  applyRotationTranslation(point);
349  synth.push_back(point) ;
350  }
351  }
352  d_synthvolume.endEdit();
353  }

356  float scale = (float)d_scale.getValue()/100.f ;

359  scale*point3d[1],
