140  void checkFiltersParams () {
141  int & decimation = *d_decimation.beginEdit() ;
142  float &
143  alpha = *d_tmp_alpha.beginEdit(),
144  delta = *d_tmp_delta.beginEdit() ;
145  if (decimation >= 8) decimation = 8 ;
146  if (decimation <= 2) decimation = 2 ;
147  if ((int)(alpha*1e2) > 100) alpha = 1. ;
148  if ((int)(alpha*1e2) < 1) alpha = .01 ;
149  if ((int)delta > 100) delta = 100 ;
150  if ((int)delta < 1) delta = 1 ;
151  d_decimation.endEdit();
152  d_tmp_alpha.endEdit();
153  d_tmp_delta.endEdit();
154  this->decimation.set_option(RS2_OPTION_FILTER_MAGNITUDE, d_decimation.getValue());
155  this->temporal.set_option(RS2_OPTION_FILTER_SMOOTH_ALPHA, d_tmp_alpha.getValue());
156  this->temporal.set_option(RS2_OPTION_FILTER_SMOOTH_DELTA, d_tmp_delta.getValue());
157  }

165  void handleEvent(sofa::core::objectmodel::Event* event) override {

172  calib_imagelist.push_back(d_color.getValue().getImage());
173  }
174  else if (ev->getKey() == 'z' || ev->getKey() == 'Z') {
175  if (calib_imagelist.size() > 0) calib_imagelist.pop_back();

178  // save

180  int i = 0 ;

182  std::string imagepath = path + "/" + std::to_string(i++) + ".png" ;

200  void writeIntrinsics (std::string filename, const rs2_intrinsics &cam_intrinsics) {
201  // for exporting intrinsics as sofa data
202  defaulttype::Vector4 & intrinsics = *d_intrinsicParameters.beginEdit() ;
203  intrinsics = defaulttype::Vector4 (
204  cam_intrinsics.fx,
205  cam_intrinsics.fy,
206  cam_intrinsics.ppx, // x0
207  cam_intrinsics.ppy // y0
208  ) ;
209  d_intrinsicParameters.endEdit();
210  std::cout << "intrinsics : " << d_intrinsicParameters.getValue() << std::endl ;
211 
212  // export intrinsics for later use (offline reproj)
213  std::FILE* filestream = std::fopen(filename.c_str(), "wb") ;
214  if (filestream == NULL) {
215  std::cerr << "Check rights on intrins.log file" << std::endl ;
216  return ;
217  }
218  std::fwrite(&cam_intrinsics.width, sizeof(int), 1, filestream) ;
219  std::fwrite(&cam_intrinsics.height, sizeof(int), 1, filestream) ;
220  std::fwrite(&cam_intrinsics.ppx, sizeof(float), 1, filestream) ;
221  std::fwrite(&cam_intrinsics.ppy, sizeof(float), 1, filestream) ;
222  std::fwrite(&cam_intrinsics.fx, sizeof(float), 1, filestream) ;
223  std::fwrite(&cam_intrinsics.fy, sizeof(float), 1, filestream) ;
224  std::fwrite(&cam_intrinsics.model, sizeof(rs2_distortion), 1, filestream) ;
225  std::fwrite(cam_intrinsics.coeffs, sizeof(float), 5, filestream) ;
226  std::fclose(filestream) ;
227  }

243  rs2::frameset wait_for_frame(rs2::pipeline & pipe) {
244  rs2::align align(RS2_STREAM_COLOR);
245  rs2::frameset frameset;
246  while (
247  (!frameset.first_or_default(RS2_STREAM_DEPTH) ||
248  !frameset.first_or_default(RS2_STREAM_COLOR))
249  ) {
250  frameset = pipe.wait_for_frames();
251  }
252  rs2::frameset processed = align.process(frameset);
253 
254  return processed;
