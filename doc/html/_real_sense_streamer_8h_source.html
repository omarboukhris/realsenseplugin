114  void handleEvent(sofa::core::objectmodel::Event* event) override {
115  if (sofa::core::objectmodel::KeypressedEvent * ev = dynamic_cast<sofa::core::objectmodel::KeypressedEvent*>(event)){
116  if (ev->getKey() == 'I' || ev->getKey() == 'i') {
117  calib_imagelist.push_back(d_color.getValue().getImage());
118  }
119  else if (ev->getKey() == 'z' || ev->getKey() == 'Z') {
120  if (calib_imagelist.size() > 0) calib_imagelist.pop_back();
121  }
122  else if (ev->getKey() == 's' || ev->getKey() == 'S') {
123  // save

125  int i = 0 ;
126  for (const auto & image : calib_imagelist) {
127  std::string imagepath = path + "/" + std::to_string(i) + ".png" ;
128  cv::imwrite(imagepath, image) ;
129  }
130  }

145  void writeIntrinsics (std::string filename, const rs2_intrinsics &cam_intrinsics) {
146  // for exporting intrinsics as sofa data
147  defaulttype::Vec4f intrinsics (
148  cam_intrinsics.fx,
149  cam_intrinsics.fy,
150  cam_intrinsics.ppx, // x0
151  cam_intrinsics.ppy // y0
152  ) ;
153  d_intrinsicParameters.setValue(intrinsics);
154  std::cout << "intrinsics : " << d_intrinsicParameters.getValue() << std::endl ;
155 
156  // export intrinsics for later use (offline reproj)
157  std::FILE* filestream = std::fopen(filename.c_str(), "wb") ;
158  if (filestream == NULL) {
159  std::cerr << "Check rights on intrins.log file" << std::endl ;
160  return ;
161  }
162  std::fwrite(&cam_intrinsics.width, sizeof(int), 1, filestream) ;
163  std::fwrite(&cam_intrinsics.height, sizeof(int), 1, filestream) ;
164  std::fwrite(&cam_intrinsics.ppx, sizeof(float), 1, filestream) ;
165  std::fwrite(&cam_intrinsics.ppy, sizeof(float), 1, filestream) ;
166  std::fwrite(&cam_intrinsics.fx, sizeof(float), 1, filestream) ;
167  std::fwrite(&cam_intrinsics.fy, sizeof(float), 1, filestream) ;
168  std::fwrite(&cam_intrinsics.model, sizeof(rs2_distortion), 1, filestream) ;
169  std::fwrite(cam_intrinsics.coeffs, sizeof(float), 5, filestream) ;
170  std::fclose(filestream) ;
171  }

178  rs2::frameset wait_for_frame(rs2::pipeline & pipe) {
179  rs2::align align(RS2_STREAM_COLOR);
180  rs2::frameset frameset;
181  while (
182  (!frameset.first_or_default(RS2_STREAM_DEPTH) ||
183  !frameset.first_or_default(RS2_STREAM_COLOR))

185  frameset = pipe.wait_for_frames();
186  }
187  rs2::frameset processed = align.process(frameset);
188 
189  return processed;
