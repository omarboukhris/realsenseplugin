135  void checkFiltersParams () {
136  int & decimation = *d_decimation.beginEdit() ;
137  float &
138  alpha = *d_tmp_alpha.beginEdit(),
139  delta = *d_tmp_delta.beginEdit() ;
140  if (decimation >= 8000) decimation = 8000 ;
141  if (decimation <= 8000) decimation = 0 ;
142  if ((int)(alpha*1e5) >= 10000) alpha = 1. ;
143  if ((int)(alpha*1e5) <= 0) alpha = 0. ;
144  if ((int)delta >= 100) delta = 100. ;
145  if ((int)delta <= 1) delta = 1. ;
146  d_decimation.endEdit();
147  d_tmp_alpha.endEdit();
148  d_tmp_delta.endEdit();
149  this->decimation.set_option(RS2_OPTION_FILTER_MAGNITUDE, d_decimation.getValue());
150  this->temporal.set_option(RS2_OPTION_FILTER_SMOOTH_ALPHA, d_tmp_alpha.getValue());
151  this->temporal.set_option(RS2_OPTION_FILTER_SMOOTH_DELTA, d_tmp_delta.getValue());
152  }

189  void writeIntrinsics (std::string filename, const rs2_intrinsics &cam_intrinsics) {
190  // for exporting intrinsics as sofa data
191  defaulttype::Vec4f intrinsics (
192  cam_intrinsics.fx,
193  cam_intrinsics.fy,
194  cam_intrinsics.ppx, // x0
195  cam_intrinsics.ppy // y0
196  ) ;
197  d_intrinsicParameters.setValue(intrinsics);
198  std::cout << "intrinsics : " << d_intrinsicParameters.getValue() << std::endl ;
199 
200  // export intrinsics for later use (offline reproj)
201  std::FILE* filestream = std::fopen(filename.c_str(), "wb") ;
202  if (filestream == NULL) {
203  std::cerr << "Check rights on intrins.log file" << std::endl ;
204  return ;
205  }
206  std::fwrite(&cam_intrinsics.width, sizeof(int), 1, filestream) ;
207  std::fwrite(&cam_intrinsics.height, sizeof(int), 1, filestream) ;
208  std::fwrite(&cam_intrinsics.ppx, sizeof(float), 1, filestream) ;
209  std::fwrite(&cam_intrinsics.ppy, sizeof(float), 1, filestream) ;
210  std::fwrite(&cam_intrinsics.fx, sizeof(float), 1, filestream) ;
211  std::fwrite(&cam_intrinsics.fy, sizeof(float), 1, filestream) ;
212  std::fwrite(&cam_intrinsics.model, sizeof(rs2_distortion), 1, filestream) ;
213  std::fwrite(cam_intrinsics.coeffs, sizeof(float), 5, filestream) ;
214  std::fclose(filestream) ;
215  }
